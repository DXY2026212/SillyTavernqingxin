    function safeGetValue(obj, path, defaultValue = null) {
        const keys = path.replace(/\[/g, '.').replace(/\]/g, '').split('.');
        let result = obj;
        for (const key of keys) {
            if (result === null || result === undefined) return defaultValue;
            result = result[key];
        }
        return result === undefined ? defaultValue : result;
    }

    function updateProperty(statData, deltaData, basePath, valueElId, descElId, defaultValue = 'N/A') {
        const valueEl = document.getElementById(valueElId);
        const descEl = document.getElementById(descElId);
        if (!valueEl) return;

        const statArray = safeGetValue(statData, basePath, [defaultValue, ""]);
        const finalValue = statArray[0];
        const staticDescription = statArray[1];

        const deltaString = safeGetValue(deltaData, basePath);
        let dynamicDescription = "";
        if (deltaString) {
            const match = String(deltaString).match(/\((.*)\)/);
            if (match && match[1]) {
                dynamicDescription = match[1];
            }
        }

        const finalDescription = dynamicDescription || staticDescription;
        valueEl.innerText = finalValue;
        if (descEl) {
            descEl.innerText = finalDescription ? `(${finalDescription})` : "";
        }
    }

    function updateProgressBar(barId, valueMainId, valueDescId, valueArray, min, max, positiveColorVar, negativeColorVar) {
        const progressBar = document.getElementById(barId);
        const valueMainDisplay = document.getElementById(valueMainId);
        const valueDescDisplay = document.getElementById(valueDescId);
        if (!valueMainDisplay || !progressBar || !Array.isArray(valueArray)) return;

        const numericValue = parseFloat(valueArray[0]) || 0;
        const descriptionText = valueArray[1] || "";
        valueMainDisplay.innerText = numericValue;
        if (valueDescDisplay) {
            valueDescDisplay.innerText = descriptionText ? `(${descriptionText})` : "";
        }
        
        const range = max - min;
        const percentage = range === 0 ? 100 : ((numericValue - min) / range) * 100;
        progressBar.style.width = `${Math.max(0, Math.min(100, percentage))}%`;
        progressBar.style.background = numericValue >= 0 ? `var(${positiveColorVar})` : `var(${negativeColorVar})`;
    }

    function updateDisplay(message) {
        try {
            const statData = safeGetValue(message, 'data.stat_data');
            const deltaData = safeGetValue(message, 'data.delta_data', {});
            if (!statData) return;

            updateProgressBar(
